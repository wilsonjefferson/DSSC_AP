\subsection{Concurrency}
\begin{frame}[fragile]
  \frametitle{We will not see the concurrency library \frownie }
\begin{lstlisting}
  int main(){
    // f and g are independent
    f();
    g();
  }
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]
  \frametitle{We will not see the concurrency library  \frownie }
\begin{lstlisting}
  #include <thread>
    
  int main(){
    // f and g are independent
    std::thread t{ f };
    g();
    t.join();
  }
\end{lstlisting}

\end{frame}



\begin{frame}[fragile]
  \frametitle{We will not see the concurrency library \frownie }
\begin{lstlisting}
  #include <future>
    
  int main(){
    // f and g are independent
    auto from_f = std::async( f );
    auto from_g = g();
    ...
    complicated( from_g, from_f.get() );
  }
\end{lstlisting}
\end{frame}
